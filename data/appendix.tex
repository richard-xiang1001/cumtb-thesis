\section{优化算法}

本节提供算法的详细伪代码。

\begin{algorithm}[htbp]
  \caption{梯度下降算法}
  \label{alg:appendix-gd}
  \begin{algorithmic}[1]
    \REQUIRE 学习率 $\alpha$，初始参数 $\theta_0$
    \ENSURE 优化后的参数 $\theta^*$
    \STATE 初始化 $t \leftarrow 0$
    \WHILE{未收敛}
      \STATE 计算梯度 $g_t \leftarrow \nabla_\theta \mathcal{L}(\theta_t)$
      \STATE 更新参数 $\theta_{t+1} \leftarrow \theta_t - \alpha g_t$
      \STATE $t \leftarrow t + 1$
    \ENDWHILE
    \RETURN $\theta^*$
  \end{algorithmic}
\end{algorithm}
